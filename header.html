// header-footer.js // Inyecta header y footer en #header-placeholder / #footer-placeholder (function () { const inject = async (id, url) => { const host = document.getElementById(id); if (!host) return; try { const res = await fetch(url, { cache: 'no-store' }); if (!res.ok) throw new Error(res.status); const html = await res.text(); host.outerHTML = html; // reemplaza el placeholder } catch (e) { console.warn('No se pudo cargar', url, e); } }; // Ejecutar al cargar document.addEventListener('DOMContentLoaded', () => { inject('header-placeholder', 'header.html'); inject('footer-placeholder', 'footer.html'); }); })();
